# Install with:  pip install -r requirements.txt

# NOTE: torch and transformers are intentionally NOT listed. The
# cross-encoder reranker runs on onnxruntime via fastembed (the same
# runtime the bge embedding model uses), so torch's ~200MB CPU build
# (and the Windows c10.dll / WinError 1114 load failures it triggered)
# is eliminated entirely.

--extra-index-url https://pypi.org/simple/


# --- MCP framework (official Python SDK, exposes FastMCP) ---
mcp>=1.2.0

# --- HTTP / Jama REST client ---
requests>=2.31.0
urllib3>=2.0.0

# --- Storage engine ---
# sqlite-vec ships prebuilt wheels on PyPI / GitHub releases.
sqlite-vec>=0.1.6
# FTS5 is bundled with CPython's sqlite3 module (no extra package needed).

# --- LlamaIndex (primary RAG framework: SentenceSplitter + Document/TextNode
#     for chunking; Multi-Query expansion is done by the MCP LLM client) ---
llama-index-core>=0.11.0

# --- HTML cleaning for Jama rich-text fields ---
beautifulsoup4>=4.12.0
lxml>=5.0.0

# --- Local embedding + cross-encoder reranker (both ONNX via fastembed) ---
# fastembed serves BOTH the bge-small-en-v1.5 embedding model AND the
# ms-marco-MiniLM-L-6-v2 cross-encoder reranker. TextCrossEncoder (the
# reranker API) is stable from 0.7+, so pin the floor there.
fastembed>=0.7.0

# onnxruntime: shared by the embedding and reranker models. Pin to a version
# whose Windows DLLs don't require the new VC++ Runtime. fastembed's constraint
# is Python-version-dependent:
#   py3.10-3.12 -> >=1.17.0, !=1.20.0   (1.20.1 is allowed and stable)
#   py3.13+     -> >1.21.0              (must use a newer build)
# 1.27.0 (what fastembed 0.8.0 pulls on py3.13) triggers the same VC++ DLL
# failure as CUDA torch on Windows, so for py3.13+ we pin 1.21.1 (the lowest
# satisfying >1.21) instead of letting it float.
onnxruntime==1.20.1; python_version < "3.13"
onnxruntime==1.21.1; python_version >= "3.13"

# --- Background incremental sync scheduler ---
APScheduler>=3.10.0
# APScheduler 3.x imports pkg_resources, which was removed in setuptools>=81.
setuptools<81

# --- Utilities ---
python-dotenv>=1.0.0
